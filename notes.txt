
It may be necessary to create a synchronised class to access FCGX_PutStr and
FCGX_PutChar.

Something like this

synchronized SFCGI_OutStream
{
  void put(const char[] c, FCGX_Stream *stream)
  {
    FCGX_PutStr(c.ptr, c.length, stream);
  }

  void put(const char c, FCGX_Stream *stream)
  {
    FCGX_PutChar(c, stream);
  }
}

shared SFCGI_OutStream fcgi_os; // create in FCGI_loop.

class FCGI_OutStream
{
  FCGX_Stream* stream;

  this(FCGX_Stream* _stream)
  {
    stream = _stream;
  }

  void put(const char[] c)
  {
    fcgi_os.put(c, stream);
  }

  void put(const char c)
  {
    fcgi_os.put(c, stream);
  }
}
